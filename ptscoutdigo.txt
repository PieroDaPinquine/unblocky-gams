    <script>
      const statusElement = document.getElementById('status');
      const progressFill = document.getElementById('progress-fill');
      const loader = document.getElementById('loader');
      
      let totalBytes = 315181892;
      let loadedBytes = 0;

      // Fast streaming fetch
      async function fetchWithProgress(url) {
        const response = await fetch(url);
        const reader = response.body.getReader();
        let receivedLength = 0;
        let chunks = []; 

        while(true) {
          const {done, value} = await reader.read();
          if (done) break;
          chunks.push(value);
          receivedLength += value.length;
          loadedBytes += value.length;
          
          // Update UI on animation frame for smoothness
          requestAnimationFrame(() => {
            const pct = Math.min((loadedBytes / totalBytes) * 100, 100).toFixed(1);
            statusElement.textContent = `DOWNLOADING DATA: ${pct}%`;
            progressFill.style.width = pct + "%";
          });
        }

        let allChunks = new Uint8Array(receivedLength);
        let position = 0;
        for(let chunk of chunks) {
          allChunks.set(chunk, position);
          position += chunk.length;
        }
        return allChunks.buffer;
      }

      (async () => {
        try {
          // Chunk Logic
          const parts = Array.from({length: 16}, (_, i) => `game.unx.part${i+1}`);
          const buffers = await Promise.all(parts.map(p => fetchWithProgress(p)));
          const blobUrl = URL.createObjectURL(new Blob(buffers));

          window.gameUnxUrl = blobUrl;

          // Optimized Proxy for XHR/Fetch
          const originalFetch = window.fetch;
          window.fetch = (url, options) => {
            if (typeof url === 'string' && url.includes("game.unx")) url = window.gameUnxUrl;
            return originalFetch(url, options);
          };

          // Inject Scripts
          const script1 = document.createElement("script");
          script1.src = "index.js";
          document.body.appendChild(script1);

          script1.onload = () => {
            const script2 = document.createElement("script");
            script2.src = "runner.js";
            script2.async = true;
            document.body.appendChild(script2);
            loader.style.display = 'none';
            document.getElementById('canvas').classList.add('active');
          };
          
        } catch (e) {
          statusElement.textContent = "CRITICAL ERROR: " + e.message;
        }
      })();
